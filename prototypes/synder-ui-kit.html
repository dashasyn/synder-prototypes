How to Use

Include the CSS file in any prototype HTML. All tokens and components are ready.

<!-- Add to your prototype <head> -->
<link href="https://fonts.googleapis.com/css2?family=Roboto:wght@400;500;700&display=swap" rel="stylesheet">
<link rel="stylesheet" href="synder-ui-kit.css">

<!-- Basic page structure -->
<div class="app-shell">
  <aside class="sidebar">...</aside>
  <div class="app-main">
    <div class="topbar">...</div>
    <div class="content">...</div>
  </div>
</div>

<!-- Use CSS variables for custom styles -->
<style>
  .my-custom { 
    color: var(--text-1); 
    border: 1px solid var(--border);
    border-radius: var(--r);
  }
</style>
